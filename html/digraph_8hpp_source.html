104Matrix<T, Dynamic, 1> getOneDimNullspaceFromSVD(const Ref<const Matrix<T, Dynamic, Dynamic> >& A)
105{
106 // Perform a singular value decomposition of A, only computing V in full
107 Eigen::BDCSVD<Matrix<T, Dynamic, Dynamic> > svd(A, Eigen::ComputeFullV);
108
109 // Return the column of V corresponding to the least singular value of A
110 // (always given last in the decomposition)
111 Matrix<T, Dynamic, 1> singular = svd.singularValues();
112 Matrix<T, Dynamic, Dynamic> V = svd.matrixV();
113 return V.col(singular.rows() - 1);
114}

128template <typename T>
129Matrix<T, Dynamic, Dynamic> getNullspaceFromSVD(const Ref<const Matrix<T, Dynamic, Dynamic> >& A, const T tol)
130{
131 // Perform a singular value decomposition of A, only computing V in full
132 Eigen::BDCSVD<Matrix<T, Dynamic, Dynamic> > svd(A, Eigen::ComputeFullV);
133
134 // Initialize nullspace basis matrix
135 Matrix<T, Dynamic, Dynamic> nullmat;
136 unsigned ncols = 0;
137 unsigned nrows = A.cols();
138
139 // Run through the singular values of A (in ascending, i.e., reverse order) ...
140 Matrix<T, Dynamic, 1> singular = svd.singularValues();
141 Matrix<T, Dynamic, Dynamic> V = svd.matrixV();
142 unsigned ns = singular.rows();
143 unsigned j = ns - 1;
144 while (isclose<T>(singular(j), 0.0, tol) && j >= 0)
145 {
146 // ... and grab the columns of V that correspond to the zero
147 // singular values
148 ncols++;
149 nullmat.resize(nrows, ncols);
150 nullmat.col(ncols - 1) = V.col(j);
151 j--;
152 }
153
154 return nullmat;
155}

220template <typename T>
221Matrix<T, Dynamic, Dynamic> chebotarevAgaevRecurrence(const Ref<const Matrix<T, Dynamic, Dynamic> >& laplacian,
222 const Ref<const Matrix<T, Dynamic, Dynamic> >& curr,
223 const int k,
224 const SummationMethod method = NaiveSummation)
225{
226 T K(k + 1);
227 Matrix<T, Dynamic, Dynamic> product;
228 T sigma;
229 switch (method)
230 {
231 case NaiveSummation:
232 product = laplacian * curr;
233 sigma = product.trace() / K;
234 break;
235
236 case KahanSummation:
237 product = KahanSum::multiply(laplacian, curr);
238 sigma = KahanSum::trace(product) / K;
239 break;
240
241 case KBNSummation:
242 product = KBNSum::multiply(laplacian, curr);
243 sigma = KBNSum::trace(product) / K;
244 break;
245
246 default:
247 std::stringstream ss;
248 ss << "Unrecognized summation method: " << method;
249 throw std::invalid_argument(ss.str());
250 break;
251 }
252 Matrix<T, Dynamic, Dynamic> identity = Matrix<T, Dynamic, Dynamic>::Identity(laplacian.rows(), laplacian.cols());
253
254 return (sigma * identity) - product;
255}

978 Matrix<T, Dynamic, Dynamic> getLaplacian()

980 // Initialize a zero matrix with #rows = #cols = #nodes
981 Matrix<T, Dynamic, Dynamic> laplacian = Matrix<T, Dynamic, Dynamic>::Zero(this->numnodes, this->numnodes);
982
983 // Populate the off-diagonal entries of the matrix first:
984 // (i,j)-th entry is the label of the edge j -> i
985 unsigned i = 0;
986 for (auto&& v : this->order)
987 {
988 unsigned j = 0;
989 for (auto&& w : this->order)
990 {
991 if (i != j)
992 {
993 // Get the edge label for j -> i
994 if (this->edges[w].find(v) != this->edges[w].end())

997 laplacian(i,j) = label;
998 if (laplacian(i,j) < 0)
999 throw std::runtime_error("Negative edge label found");
1000 }
1001 }
1002 j++;
1003 }
1004 i++;
1005 }
1006
1007 // Populate diagonal entries as negative sums of the off-diagonal
1008 // entries in each column
1009 for (unsigned i = 0; i < this->numnodes; ++i)
1010 laplacian(i,i) = -(laplacian.col(i).sum());
1011
1012 return laplacian;

1919 }
1920 }

1926 if (t == 0)

1928 second_moments.tail(z) = forest_two_roots_sub_squared.block(1, 1, z, z).rowwise().sum();

1933 }

1940 }

1951 }

1954 second_moments.head(t) = KahanSum::rowSum(forest_two_roots_sub_squared.block(0, 0, t, t));
1955 second_moments.head(t) += KahanSum::rowSum(forest_two_roots_sub_squared.block(0, t + 1, t, z));
1956 second_moments.tail(z) = KahanSum::rowSum(forest_two_roots_sub_squared.block(t + 1, 0, z, t));
1957 second_moments.tail(z) += KahanSum::rowSum(forest_two_roots_sub_squared.block(t + 1, t + 1, z, z));

1973 second_moments.head(t) += KBNSum::rowSum(forest_two_roots_sub_squared.block(0, t + 1, t, z));
